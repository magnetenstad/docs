const calendarData = {
	entries: [],
	colors: colors,
}

const pages = dv.pages()
const notesEachDay = new Map()

for (let page of pages) {
	for (let key of [page.file.cday.toString(), page.file.htmlay.toString()]) {
		const prev = notesEachDay.get(key) || 0
		notesEachDay.set(key, prev + 1)
	}
}

for (const [key, value] of notesEachDay.entries()) {
	calendarData.entries.push({
		date: key,
		intensity: value,
		color: "green",
		content: value,
	})
}

renderHeatmapCalendar(this.container, calendarData)
